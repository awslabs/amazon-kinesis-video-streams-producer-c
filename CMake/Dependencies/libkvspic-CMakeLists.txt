cmake_minimum_required(VERSION 3.6.3)

project(libkvspic-download LANGUAGES C)

include(ExternalProject)

if (PIC_VERSION_OVERRIDE)
  set(KVSPIC_GIT_TAG "${PIC_VERSION_OVERRIDE}")
  message(WARNING "Override PIC to a custom version: ${KVSPIC_GIT_TAG}")
else()
  set(KVSPIC_GIT_TAG "v1.3.1")
endif()

message(STATUS "Using KVS PIC version: ${KVSPIC_GIT_TAG}")

# clone repo only
ExternalProject_Add(libkvspic-download
	GIT_REPOSITORY    https://github.com/awslabs/amazon-kinesis-video-streams-pic.git
	GIT_TAG           ${KVSPIC_GIT_TAG}
   	SOURCE_DIR        "${CMAKE_CURRENT_BINARY_DIR}/kvspic-src"
	BINARY_DIR        "${CMAKE_CURRENT_BINARY_DIR}/kvspic-build"
	CMAKE_ARGS
	  -DCMAKE_BUILD_TYPE=${CMAKE_BUILD_TYPE}
	CONFIGURE_COMMAND ""
	BUILD_COMMAND     ""
	INSTALL_COMMAND   ""
	TEST_COMMAND      ""
)
